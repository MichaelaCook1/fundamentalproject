133class LazyProxy(object): 

134 """Class for proxy objects that delegate to a specified function to evaluate 

135 the actual object. 

136 

137 >>> def greeting(name='world'): 

138 ... return 'Hello, %s!' % name 

139 >>> lazy_greeting = LazyProxy(greeting, name='Joe') 

140 >>> print(lazy_greeting) 

141 Hello, Joe! 

142 >>> u' ' + lazy_greeting 

143 u' Hello, Joe!' 

144 >>> u'(%s)' % lazy_greeting 

145 u'(Hello, Joe!)' 

146 

147 This can be used, for example, to implement lazy translation functions that 

148 delay the actual translation until the string is actually used. The 

149 rationale for such behavior is that the locale of the user may not always 

150 be available. In web applications, you only know the locale when processing 

151 a request. 

152 

153 The proxy implementation attempts to be as complete as possible, so that 

154 the lazy objects should mostly work as expected, for example for sorting: 

155 

156 >>> greetings = [ 

157 ... LazyProxy(greeting, 'world'), 

158 ... LazyProxy(greeting, 'Joe'), 

159 ... LazyProxy(greeting, 'universe'), 

160 ... ] 

161 >>> greetings.sort() 

162 >>> for greeting in greetings: 

163 ... print(greeting) 

164 Hello, Joe! 

165 Hello, universe! 

166 Hello, world! 

167 """

356 # The encoding of a msgctxt and a msgid in a .mo file is

358 CONTEXT_ENCODING = '%s\x04%s'

360 def pgettext(self, context, message): 

361 """Look up the `context` and `message` id in the catalog and return the 

362 corresponding message string, as an 8-bit string encoded with the 

363 catalog's charset encoding, if known. If there is no entry in the 

364 catalog for the `message` id and `context` , and a fallback has been 

365 set, the look up is forwarded to the fallback's ``pgettext()`` 

366 method. Otherwise, the `message` id is returned. 

367 """ 

368 ctxt_msg_id = self.CONTEXT_ENCODING % (context, message) 

369 missing = object() 

370 tmsg = self._catalog.get(ctxt_msg_id, missing) 

371 if tmsg is missing: 

372 if self._fallback: 

373 return self._fallback.pgettext(context, message) 

374 return message 

375 # Encode the Unicode tmsg back to an 8-bit string, if possible 

376 if self._output_charset: 

377 return text_to_native(tmsg, self._output_charset) 

378 elif self._charset: 

379 return text_to_native(tmsg, self._charset) 

380 return tmsg

398 def npgettext(self, context, singular, plural, num): 

399 """Do a plural-forms lookup of a message id. `singular` is used as the 

400 message id for purposes of lookup in the catalog, while `num` is used to 

401 determine which plural form to use. The returned message string is an 

402 8-bit string encoded with the catalog's charset encoding, if known. 

403 

404 If the message id for `context` is not found in the catalog, and a 

405 fallback is specified, the request is forwarded to the fallback's 

406 ``npgettext()`` method. Otherwise, when ``num`` is 1 ``singular`` is 

407 returned, and ``plural`` is returned in all other cases. 

408 """ 

409 ctxt_msg_id = self.CONTEXT_ENCODING % (context, singular) 

410 try: 

411 tmsg = self._catalog[(ctxt_msg_id, self.plural(num))] 

412 if self._output_charset: 

413 return text_to_native(tmsg, self._output_charset) 

414 elif self._charset: 

415 return text_to_native(tmsg, self._charset) 

416 return tmsg 

417 except KeyError: 

418 if self._fallback: 

419 return self._fallback.npgettext(context, singular, plural, num) 

420 if num == 1: 

421 return singular 

422 else: 

423 return plural

552 def load(cls, dirname=None, locales=None, domain=None): 

553 """Load translations from the given directory. 

554 

555 :param dirname: the directory containing the ``MO`` files 

556 :param locales: the list of locales in order of preference (items in 

557 this list can be either `Locale` objects or locale 

558 strings) 

559 :param domain: the message domain (default: 'messages') 

560 """ 

561 if locales is not None: 

562 if not isinstance(locales, (list, tuple)): 

563 locales = [locales] 

564 locales = [str(locale) for locale in locales] 

565 if not domain: 

566 domain = cls.DEFAULT_DOMAIN 

567 filename = gettext.find(domain, dirname, locales) 

568 if not filename: 

569 return NullTranslations() 

570 with open(filename, 'rb') as fp: 

571 return cls(fp=fp, domain=domain)

577 def add(self, translations, merge=True): 

578 """Add the given translations to the catalog. 

579 

580 If the domain of the translations is different than that of the 

581 current catalog, they are added as a catalog that is only accessible 

582 by the various ``d*gettext`` functions. 

583 

584 :param translations: the `Translations` instance with the messages to 

585 add 

586 :param merge: whether translations for message domains that have 

587 already been added should be merged with the existing 

588 translations 

589 """ 

590 domain = getattr(translations, 'domain', self.DEFAULT_DOMAIN) 

591 if merge and domain == self.domain: 

592 return self.merge(translations) 

593 

594 existing = self._domains.get(domain) 

595 if merge and existing is not None: 

596 existing.merge(translations) 

597 else: 

598 translations.add_fallback(self) 

599 self._domains[domain] = translations 

600 

601 return self

603 def merge(self, translations): 

604 """Merge the given translations into the catalog. 

605 

606 Message translations in the specified catalog override any messages 

607 with the same identifier in the existing catalog. 

608 

609 :param translations: the `Translations` instance with the messages to 

610 merge 

611 """ 

612 if isinstance(translations, gettext.GNUTranslations): 

613 self._catalog.update(translations._catalog) 

614 if isinstance(translations, Translations): 

615 self.files.extend(translations.files) 

616 

617 return self
